184  void FlowMap(State &dx, double t, double k, const State &x,
185  const Input &u, const Params &p) const override {
186  dx[0] = x[1];
187  dx[1] = -p[p_idx::k1]/p[p_idx::m1] * x[0] - p[p_idx::c1]/p[p_idx::m1] * x[1] + u[0];
188  dx[2] = x[3];
189  dx[3] = -p[p_idx::k2]/p[p_idx::m2] * (x[2] - x[0]) - p[p_idx::c2]/p[p_idx::m2] * (x[3] - x[1]);
190  };

212  void JumpMap(State &xp, double t, double k, const State &x, const Input &u, const Params &p) const override {
213  hybrid::math::copy(xp, x);
214  xp[0] = x[0];
215  xp[1] = ((p[p_idx::m1] - p[p_idx::m2]) * x[1] + 2 * p[p_idx::m2] * x[3])/(p[p_idx::m1] + p[p_idx::m2]);
216  xp[2] = x[2];
217  xp[3] = ((p[p_idx::m2] - p[p_idx::m1]) * x[3] + 2 * p[p_idx::m1] * x[1])/(p[p_idx::m1] + p[p_idx::m2]);
218  };

239  bool JumpSet(double t, double k, const State &x, const Input &u,
240  const Params &p) const override {
241  if (((x[2] - x[0]) >= p[p_idx::d]/2) && ((x[3] - x[1]) >= 0))
242  return true;
243  if (((x[2] - x[0]) <= -p[p_idx::d]/2) && ((x[3] - x[1]) <= 0))
244  return true;
245  return false;

82  int solve(std::array< REAL_T, dim_x > &x, Args... data) {
83  int iter = 0;
84  std::array< REAL_T, dim_x > wv_c;
85  std::array< REAL_T, dim_x > wv_d;
86  std::fill(std::begin(wv_c), std::end(wv_c), 0.0);
87  std::fill(std::begin(wv_d), std::end(wv_d), 0.0);
88 
89  while (iter <= this->max_iter) {
90  this->g(this->wv_b, x, data...);
91  if (hybrid::math::norm(this->wv_b) < this->tol) return iter; // Exit because |g(x)| approx 0
92  hybrid::math::scale(this->wv_b, -1.0);
93  this->dg(this->wv_a, x, data...);
94  hybrid::math::solve< REAL_T, dim_g, dim_x >(this->wv_a, this->wv_b);
95  hybrid::math::sum(x, this->wv_b);
96 
97  // Evaluating relative x tolerance |δ x+ - δx | / xtol
98  for (std::size_t i = 0; i < dim_x; i++) {
99  wv_d[i] = (this->wv_b[i] - wv_c[i]) / this->xtol;
100  wv_d[i] = wv_d[i] >= 0 ? wv_d[i] : -wv_d[i];
101  }
102  if (hybrid::math::element_compare(wv_c, wv_d)) {
103  throw hybrid::XTolException("Update for target variable is less than tolerance"); // Exit because |dx| approx 0
104  }
105  hybrid::math::copy(wv_c, this->wv_b);
106  iter++;
107  }
108  throw hybrid::MaxIterException("Maximum number of iteration reached without solution");
109  }
110  };

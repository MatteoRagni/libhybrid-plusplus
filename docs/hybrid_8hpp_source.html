256  void step(std::array< REAL_T, dim_x > &xp, REAL_T &t, REAL_T &k, const std::array< REAL_T, dim_x > &x,
257  const std::array< REAL_T, dim_u > &u, const std::array< REAL_T, dim_p > &p, Args... data) {
258  if (t > this->time_limit || k > this->jumps_limit) {
259  throw std::runtime_error("Time or Jump limits reached");
260  }
261 
262  if (this->should_jump(t, k, x, u, p, data...)) {
263  k++;
264  this->JumpMap(xp, t, k, x, u, p, data...);
265  } else {
266  this->next(xp, t, x, u, p, k, data...);
267  }
268  };

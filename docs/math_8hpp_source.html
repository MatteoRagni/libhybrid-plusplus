222  template < std::size_t rows, std::size_t cols, std::size_t dimb >
223  void solve(std::array< double, rows * cols > &A, std::array< double, dimb > &b) {
224  lapack_int result =
225  DGELS_CALL(LAPACK_COL_MAJOR, 'N', static_cast< lapack_int >(rows), static_cast< int >(cols), 1, A.data(),
226  static_cast< lapack_int >(rows), b.data(), static_cast< lapack_int >(dimb));
227  if (result > 0)
228  throw IllegalMatrix("[HYBRID::MATH::SOLVE] An illegal matrix was passed to solve");
229  if (result < 0)
230  throw SingularMatrix("[HYBRID::MATH::SOLVE] A singular matrix was passed to solve");
231  }
